**读取并显示摄像头数据参考代码如下：**

```python
import cv2

def main():
    rtsp_url = "rtsp://10.126.59.120:8554/live/live1"
    # 使用FFMPEG 后端并开启TCP 传输，减少花屏/延迟
    cap = cv2.VideoCapture(rtsp_url, cv2.CAP_FFMPEG)
    cap.set(cv2.CAP_PROP_BUFFERSIZE, 1)
    
    if not cap.isOpened():
        print(f"无法打开视频流: {rtsp_url}")
        return
    
    print(f"成功连接到视频流: {rtsp_url}")
    print("按 q 退出，按 s 截图保存为 snapshot.jpg")
    
    while True:
        ret, frame = cap.read()
        if not ret:
            print("读取帧失败，流可能已断开")
            break
        
        cv2.imshow("RTSP Stream", frame)
        key = cv2.waitKey(1) & 0xFF
```

**识别并显示识别后的画面参考代码（以 YOLO 为例）：**

```python
import cv2
from ultralytics import YOLO

def main():
    rtsp_url = "rtsp://10.126.59.120:8554/live/live1"

    # 加载 YOLOv8 预训练模型（替换成自己的模型）
    model = YOLO("yolov8n.pt")

    # 读取 RTSP 流
    cap = cv2.VideoCapture(rtsp_url, cv2.CAP_FFMPEG)
    cap.set(cv2.CAP_PROP_BUFFERSIZE, 1)

    if not cap.isOpened():
        print(f"无法打开视频流: {rtsp_url}")
        return

    print(f"成功连接到视频流: {rtsp_url}")
    print("按 q 退出")

    while True:
        ret, frame = cap.read()
        if not ret:
            print("读取帧失败，流可能已断开")
            break

        # YOLO 推理（stream=True 更省内存）
        results = model(frame, verbose=False)

        # 在原图上绘制识别框和标签
        annotated = results[0].plot()

        cv2.imshow("YOLO Detection", annotated)

        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

### 2 实时识别+推送

在实时识别+推送前，需要安装流媒体服务器 mediamtx 与推流工具 ffmpeg，请进入下列网站下载与系统对应版本的工具。以 Windows 为例，其中 mediamtx 下载之后，解压至指定目录，并运行 mediamtx.exe 即可（运行之后若有提示网络请允许，另外弹出的终端不能关闭）。而 ffmpeg 则下载完解压至指定目录即可（若后续调用时提示版本问题，请更新相关依赖，或者使用旧版本）。

- 流媒体服务器 mediamtx
- 推流工具 ffmpeg

**实时识别+推流代码：**

```python
import os
import subprocess
import sys
import cv2
from ultralytics import YOLO

# 输入/输出 RTSP 流地址
SRC_URL = "rtsp://10.126.59.120:8554/live/live1"  # 拉流地址
DST_URL = "rtsp://127.0.0.1:8554/video30"         # 推流地址
```

## 第 4 页

```python
# 推流参数（按目标服务器的期望调整）
FPS = 25
BITRATE = "2M"
FFMPEG_BIN = "/opt/homebrew/bin/ffmpeg"  # 这里替换成自己 ffmpeg 的路径

def start_ffmpeg(width: int, height: int, fps: int) -> subprocess.Popen:
    """启动 ffmpeg 子进程，从 stdin 读取 rawvideo 帧并推送到 DST_URL。
    需要本机已安装 ffmpeg，并已在目标地址运行一个支持 publish 的 RTSP 服务器（如 mediamtx）。
    """
    command = [
        FFMPEG_BIN,
        "-y",
        "-loglevel", "error",
        "-f", "rawvideo",
        "-vcodec", "rawvideo",
        "-pix_fmt", "bgr24",
        "-s", f"{width}x{height}",
        "-r", str(fps),
        "-i", "-",          # 从 stdin 读
        "-c:v", "libx264",
        "-pix_fmt", "yuv420p",
        "-preset", "ultrafast",
        "-tune", "zerolatency",
        "-b:v", BITRATE,
        "-f", "rtsp",
        DST_URL,
    ]
    # 让 ffmpeg 的 stderr 直接透传到当前终端，方便排查推流错误
    return subprocess.Popen(command, stdin=subprocess.PIPE, stderr=sys.stderr)

def main():
    model = YOLO("yolov8n.pt")  # 模型替换成自己的
    cap = cv2.VideoCapture(SRC_URL, cv2.CAP_FFMPEG)
    cap.set(cv2.CAP_PROP_BUFFERSIZE, 1)

    if not cap.isOpened():
        print(f"无法打开视频流: {SRC_URL}", file=sys.stderr)
        return

    print(f"成功连接到视频流: {SRC_URL}")
    width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH)) or 1280
    height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT)) or 720
    print(f"分辨率: {width}x{height}, 准备推流到 {DST_URL}")

    proc = start_ffmpeg(width, height, FPS)
    print("开始 YOLO 实时识别并推流，按 q 退出（若启用了本地显示）")

    try:
        while True:
            ret, frame = cap.read()
            if not ret:
                print("读取帧失败，流可能已断开", file=sys.stderr)
                break
```
